def verificacao(df):
  # Verificação unitária dos dados presentes em comorbidades
  selecao = df['comorbidades'].str.split('/',10)
  acum_0 = []
  for i in selecao:
    try:
      for ii in i:
        ii = unidecode(ii.strip().strip())
        if ii in acum_0:
          continue
        else:
          acum_0.append(ii)
    except:
      continue

  return sorted(acum_0) # retorna em ordem alfabetica.

def faixa_et(x):
  if x < 12:
    return 'criança'
  elif (x >= 12) & (x < 18):
    return 'adolescente'
  elif (x >= 18) & (x < 29):
    return 'jovem'
  elif (x >= 29) & (x < 60):
    return 'adulto'
  elif x >= 60:
    return 'idoso'


dados['faixa_et'] = dados['idade'].apply(lambda x: faixa_et(x))

# As funções abaixo foram desenvolvidas para modificar certos elementos presentes nos dados textuais.
def substituicao(words,word, df):
  for palav in words:
    df['comorbidades'] = df['comorbidades'].apply(lambda x: x.replace(palav,word))

  return df

def sub_0(x):
  volt = []
  for i in x:
    i = i.strip().strip()
    i = i + '.'
    i = i.replace('cr.','(cr).').replace('.','')
    volt.append(i)
  return '/'.join(volt)

def grau_doenca(df, coluna):
  """
  df : DataFrame 
    Dataset contendo texto
  coluna : string
    Coluna de texto a ser feito o tf-idf

  """
  tfidf = TfidfVectorizer(analyzer='word',stop_words = stopwords) 
  tfidf_matriz = tfidf.fit_transform(df[coluna]) # Remover stopwords e criar uma matriz sparsa 
  tf_idf = pd.DataFrame(
    tfidf_matriz.toarray(),
    columns=tfidf.get_feature_names(),
    index=df.index
  )
  tf_idf = pd.DataFrame(tf_idf.sum(0).sort_values(ascending=False).reset_index())
  tf_idf.columns = ['Doenças','Peso']
  return tf_idf

    return 'Quinta'
  elif x == 'Friday':
